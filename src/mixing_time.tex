\begin{frame}{Passeio}
    \begin{block}{O que é um passeio (walk)? ##AJUSTAR A DEFINIÇÃO}
        Um \textbf{passeio} em um grafo é uma sequência alternada de vértices e arestas
        \[
            v_0, e_1, v_1, e_2, v_2, \dots, e_k, v_k
        \]
        onde cada aresta $e_i$ conecta o vértice $v_{i-1}$ ao vértice $v_i$.
        \medskip

        \alert{Importante:} um passeio \textbf{pode repetir} vértices.
        No exemplo, o vértice \textcolor{red}{A COMPLETAR} aparece várias vezes (no início, no meio e no fim).
    \end{block}
\end{frame}


\begin{frame}{Passeio}
    \begin{center}  % <--- centraliza o conteúdo
        \begin{tikzpicture}[
            vertex/.style={circle, draw, minimum size=6mm, inner sep=0pt},
            scale=0.9   % opcional: ajuste o tamanho
            ]

            % --- Vértices dispostos simetricamente em torno da origem (0,0) ---
            \node[vertex] (A) at (0,0)    {A};
            \node[vertex] (B) at (3,0)    {B};
            \node[vertex] (C) at (2.5,2)  {C};
            \node[vertex] (D) at (0,3)    {D};
            \node[vertex] (E) at (-2.5,2) {E};
            \node[vertex] (F) at (-3,0)   {F};

            % --- Grafo base (preto) ---
            % Arestas do perímetro (hexágono)
            \draw (A) -- (B) -- (C) -- (D) -- (E) -- (F) -- (A);

            % Arestas extras (diagonais) para deixar o grafo mais interessante
            \draw (A) -- (C);
            \draw (A) -- (D);
            \draw (B) -- (E);

            % --- Passeio vermelho que RETORNA ao vértice A ---
            % Sequência: A -> B -> C -> A -> D -> E -> F -> A
            \draw[red, thick, ->]
                (A) -- (B) -- (C) -- (A) -- (D) -- (E) -- (F) -- (A);

        \end{tikzpicture}
    \end{center}

    \vspace{0.5cm}

    {\large \textbf{Sequência do passeio:}}
    \quad
    \texttt{\textcolor{red}{A → B → C → A → D → E → F → A}}
\end{frame}

\begin{frame}{Passeio Aleatório}
    \centering
    \begin{tikzpicture}[
        vertex/.style={circle, draw, minimum size=6mm, inner sep=0pt},
        >=stealth'
        ]

        % Vértices do K3 (triângulo equilátero)
        \node[vertex] (A) at (0,1) {A};
        \node[vertex] (B) at (-1,-0.5) {B};
        \node[vertex] (C) at (1,-0.5) {C};

        \draw (A) -- (B) node[midway, left, blue] {$\frac{1}{2}$};
        % Aresta A--C com a probabilidade escrita nela
        \draw (A) -- (C) node[midway, right, blue] {$\frac{1}{2}$};
        % Aresta B--C (sem rótulo, pois não estamos nela no exemplo)
        \draw (B) -- (C);

        % Grafo K3 (todas as arestas)
        \draw (A) -- (B) -- (C) -- (A);

        % Passeio aleatório (exemplo): A -> B -> A -> C -> A
        \draw[red, thick, ->] 
            (A) -- (B) -- (A) -- (C) -- (A);
    \end{tikzpicture}

    \vspace{0.3cm}

    {\large \textbf{Passeio aleatório:}}
    \quad
    \texttt{\textcolor{red}{A → B → A → C → A}}

    \vspace{0.2cm}

    \begin{block}{Como funciona}
        No $K_3$, cada vértice tem 2 vizinhos. A cada passo, você escolhe um deles uniformemente ao acaso:

        \medskip
        \centering
        Se você está no vértice \textcolor{blue}{A}, a probabilidade de ir para \textcolor{blue}{B} é $1/2$ e para \textcolor{blue}{C} é $1/2$.

        \smallskip
        O caminho vermelho acima é um exemplo de passeio aleatório: \texttt{A → B → A → C → A}.
    \end{block}
\end{frame}

\begin{frame}{Passeio Aleatório Preguiçoso}
    \centering
    \begin{tikzpicture}[
        vertex/.style={circle, draw, minimum size=6mm, inner sep=0pt},
        >=stealth'
        ]

        % --- Vértices do K_{2,3} (grafo bipartido completo) ---
        % Partição 1: A e B (lado esquerdo)
        \node[vertex] (A) at (-1.5,0.5) {A};
        \node[vertex] (B) at (-1.5,-0.5) {B};
        % Partição 2: C, D, E (lado direito)
        \node[vertex] (C) at (1.5,1) {C};
        \node[vertex] (D) at (1.5,0) {D};
        \node[vertex] (E) at (1.5,-1) {E};

        % --- Arestas do grafo bipartido (todas as conexões entre partições) ---
        \draw (A) -- (C) node[midway, above, sloped, blue] {$\frac{1}{2}\cdot\frac{1}{3}$};
        \draw (A) -- (D) node[midway, above, sloped, blue] {$\frac{1}{2}\cdot\frac{1}{3}$};
        \draw (A) -- (E) node[midway, above, sloped, blue] {$\frac{1}{2}\cdot\frac{1}{3}$};
        \draw (B) -- (C) node[midway, below, sloped, blue] {$\frac{1}{2}\cdot\frac{1}{3}$};
        \draw (B) -- (D) node[midway, below, sloped, blue] {$\frac{1}{2}\cdot\frac{1}{3}$};
        \draw (B) -- (E) node[midway, below, sloped, blue] {$\frac{1}{2}\cdot\frac{1}{3}$};

        % --- Probabilidade de ficar parado (self-loop simbólico) ---
        % Desenhamos um loop em cada vértice para indicar a opção "ficar"
        \draw[blue, thick] (A) edge[loop left, out=150, in=210, distance=0.8cm] node[above left] {$1/2$} (A);
        \draw[blue, thick] (B) edge[loop left, out=150, in=210, distance=0.8cm] node[below left] {$1/2$} (B);
        \draw[blue, thick] (C) edge[loop right, out=30, in=-30, distance=0.8cm] node[above right] {$1/2$} (C);
        \draw[blue, thick] (D) edge[loop right, out=30, in=-30, distance=0.8cm] node[right] {$1/2$} (D);
        \draw[blue, thick] (E) edge[loop right, out=30, in=-30, distance=0.8cm] node[below right] {$1/2$} (E);

        % --- Exemplo de passeio preguiçoso (vermelho) ---
        % Sequência: A -> (fica) A -> C -> (fica) C -> B -> E
        \draw[red, thick, ->] 
            (A) edge[loop above, out=60, in=120, distance=0.6cm] (A)  % fica em A
            (A) -- (C)
            (C) edge[loop above, out=30, in=-30, distance=0.6cm] (C) % fica em C
            (C) -- (B)
            (B) -- (E);

    \end{tikzpicture}

    \vspace{0.3cm}

    {\large \textbf{Exemplo de passeio preguiçoso:}}
    \quad
    \texttt{\textcolor{red}{A → A → C → C → B → E}}
    \smallskip
\end{frame}
