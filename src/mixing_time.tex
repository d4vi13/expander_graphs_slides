\subsection{Passeios}

%====================================================================
\begin{frame}{Passeio}
    \begin{block}{O que é um passeio?}

        Em um grafo $G = (V, E)$, um passeio é uma sequência finita alternada de vértices
        \[
            P = (v_0, v_1, \dots, v_n),
        \]
        tal que $\{v_{i-1}, v_i\} \in E \;\; \forall \;\; 1 \le i \le n$.
        Os vértices $v_0$ e $v_n$ são chamados de início e fim do passeio, respectivamente, os demais vértices são os vértices internos.
    \end{block}
\end{frame}

\subsection{Distribuição Estacionária e Uniforme}
\begin{frame}{Distribuição Estacionária e Uniforme}


    \textbf{Ideia:}

    \bigskip

    Imagine um passeio aleatório no grafo.

    \begin{itemize}
        \item No início sabemos exatamente onde está.
        \item Após alguns passos, a probabilidade de estar em cada vértice muda.
        \item Depois de muitos passos, essas probabilidades estabilizam.
    \end{itemize}

    Chamamos essa distribuição de
    \textbf{distribuição estacionária}.

    \bigskip

    \alert{Intuição: o passeio "esqueceu" onde começou.}
\end{frame}



\begin{frame}{Tempo de Mistura}
    \textbf{Pergunta:}

    \bigskip

    Quanto tempo leva para o passeio ficar próximo da
    distribuição estacionária?

    \bigskip

    \[
        \tau_{mix}(\varepsilon)
        =
        \min
        \left\{
            t :
            d_{TV}\!\left(P^t(x,\cdot),\pi\right)
            \le \varepsilon
        \right\}
    \]

    \bigskip

    \textbf{Interpretação:}

    \begin{itemize}
        \item Quanto menor o tempo de mistura,
            mais rapidamente o grafo espalha informação.
    \end{itemize}

    \bigskip

    \alert{Mistura rápida = pequeno tempo de mistura.}
\end{frame}

\subsection{O tempo de mistura e a lacuna espectral}


\begin{frame}{Lacuna Espectral}
  \begin{definition}[Lacuna Espectral]
    Seja M uma matriz e $\lambda_1\ge\lambda_2\ge\cdots\lambda_n$ os autovalores dela, a lacuna espectral será a diferença entre os dois maiores autovalores:
    \[
      \lambda_1 - \lambda_2
    \]
  \end{definition}
\end{frame}


\begin{frame}{Tempo de Mistura e Lacuna Espectral}
  \begin{theorem}[Lacuna espectral e tempo de mistura]
    \label{thm:spectral_mixing}
    Seja $G$ um grafo $d$-regular conexo, e seja $P$ a matriz de transição do passeio aleatório preguiçosa em $G$. Seja $1=\mu_1>\mu_2\geq \cdots \geq \mu_n$
    os autovalores de $P$, denotemos a lacuna espectral por:
    \[
      \gamma = 1-\mu_2.
    \]
    O tempo de mistura satisfaz
    \[
      \tau_{\mathrm{mix}}(\varepsilon)
      =
      O\!\left(\frac{1}{\gamma}\log n\right).
    \]
  \end{theorem}
\end{frame}

\begin{frame}{Lacuna Espectral}
    \textbf{Intuição:}

    \begin{itemize}
        \item \(\mu_2\) próximo de \(1\) $\Rightarrow$ mistura lenta.
        \item \(\mu_2\) pequeno $\Rightarrow$ mistura rápida.
    \end{itemize}

    \bigskip

    \[
        \boxed{\text{Gap grande } \Rightarrow \text{ mistura rápida}}
    \]
    
    \alert{
        Quanto maior a lacuna espectral,
        mais rapidamente o passeio converge para a distribuição estacionária.
    }


\end{frame}

\begin{frame}{Autovalores do Passeio Preguiçoso}
  \begin{prop}
\label{prop:p_eigenvalues_form}
    Seja G um grafo d-regular conexo, A sua matriz de adjacência com autovalores $d=\lambda_1 \ge \lambda_2 \ge\cdots\ge \lambda_n$, e P a matriz de transição do passeio aleatório preguiçosa em G. Os autovalores $\mu_i$ de P são da forma:
    \[
    \mu_i = \frac12 +\frac{\lambda_i}{2d}
    \]

    Note que os autovalores de P serão ordenados da mesma forma que os autovalores de A
\end{prop}
\end{frame}

\begin{frame}{Autovalores do Passeio Preguiçoso}
  \begin{proof}
    Note que os autovalores de A são dessa forma pois G é d-regular

    Sabe-se que $P = \frac12I + \frac1{2d}A$. Seja $\lambda_i$ um autovalor de A, então existe um vetor $v_i$ tal que $Av_i= \lambda_iv_i$, então o efeito de P em v:
    \[
      Pv = \frac12Iv + \frac1{2d}Av = \frac12v + \frac{\lambda_i}{2d}v = (\frac12 + \frac{\lambda_i}{2d})v 
    \]
    Então $\frac12 + \frac{\lambda_i}{2d}$ é autovalor de P, como P e A tem |V| de autovalores, todos são dessa forma.
  \end{proof}
\end{frame}


\begin{frame}{Lacuna espectral do Passeio Preguiçoso}
  \begin{corollary}[A lacuna espectral de P]
\label{cor:p_spectral_gap}
Seja G um grafo d-regular conexo e P a matriz de transição de um passeio
aleatório preguiçoso em G. A lacuna espectral de P é:
\[
    \frac{d-\lambda_2}{2d}
\]    
\end{corollary}
\end{frame}
\begin{frame}{Lacuna espectral do Passeio Preguiçoso}
\begin{proof}
    Como vimos os autovalores de P tem uma forma especifica e eles são ordenados, basta tomar os dois maiores autovalores:
    \[
    \mu_1 = \frac12 + \frac{\lambda_1}{2d} = \frac12 + \frac{d}{2d} = \frac12 + \frac12 = 1
    \] 
    \[
    \mu_2 = \frac12 + \frac{\lambda_2}{2d} 
    \]
    Então a lacuna será:
    \[
    \mu_1-\mu_2 = 1 - (\frac12 + \frac{\lambda_2}{2d}) =\frac12 - \frac{\lambda_2}{2d}  = \frac{d - \lambda_2}{2d}
    \]
\end{proof}
\end{frame}
