\subsection{Passeios}

%====================================================================
\begin{frame}{Passeio}
    \begin{block}{O que é um passeio?}

        Em um grafo $G = (V, E)$, um passeio é uma sequência finita alternada de vértices
        \[
            P = (v_0, v_1, \dots, v_n),
        \]
        tal que $\{v_{i-1}, v_i\} \in E \;\; \forall \;\; 1 \le i \le n$.
        Os vértices $v_0$ e $v_n$ são chamados de início e fim do passeio, respectivamente, os demais vértices são os vértices internos.
    \end{block}
\end{frame}

    \begin{center}  % <--- centraliza o conteúdo

        % Aresta A--C com a probabilidade escrita nela
        \draw (A) -- (C) node[midway, right, blue] {$\frac{1}{2}$};
        % Aresta B--C (sem rótulo, pois não estamos nela no exemplo)
        \draw (B) -- (C);

        % Grafo K3 (todas as arestas)
        \draw (A) -- (B) -- (C) -- (A);

        % Passeio aleatório (exemplo): A -> B -> A -> C -> A
        \draw[red, thick, ->] 
            (A) -- (B) -- (A) -- (C) -- (A);
    \end{tikzpicture}

    \vspace{0.3cm}

    {\large \textbf{Passeio aleatório:}}
    \quad
    \texttt{\textcolor{red}{1 → 2 → 1 → 3 → 1}}

    \vspace{0.2cm}

    \begin{block}{Como funciona?}
        No $K_3$, cada vértice tem 2 vizinhos. A cada passo, você escolhe um deles uniformemente ao acaso:

        \medskip
        \centering
        Se você está no vértice \textcolor{blue}{1}, a probabilidade de ir para \textcolor{blue}{2} é $1/2$ e para \textcolor{blue}{3} é $1/2$.
    \end{block}
\end{frame}

\subsection{Distribuição Estacionária e Uniforme}
\begin{frame}{Distribuição Estacionária e Uniforme}


    \textbf{Ideia:}

    \bigskip

    Imagine um passeio aleatório no grafo.

    \begin{itemize}
        \item No início sabemos exatamente onde está.
        \item Após alguns passos, a probabilidade de estar em cada vértice muda.
        \item Depois de muitos passos, essas probabilidades estabilizam.
    \end{itemize}

    Chamamos essa distribuição de
    \textbf{distribuição estacionária}.

    \bigskip

    \alert{Intuição: o passeio "esqueceu" onde começou.}
\end{frame}


\subsection{O tempo de mistura e a lacuna espectral}

\begin{frame}{Tempo de Mistura}
    \textbf{Pergunta:}

    \bigskip

    Quanto tempo leva para o passeio ficar próximo da
    distribuição estacionária?

    \bigskip

    \[
        \tau_{mix}(\varepsilon)
        =
        \min
        \left\{
            t :
            d_{TV}\!\left(P^t(x,\cdot),\pi\right)
            \le \varepsilon
        \right\}
    \]

    \bigskip

    \textbf{Interpretação:}

    \begin{itemize}
        \item Quanto menor o tempo de mistura,
            mais rapidamente o grafo espalha informação.
    \end{itemize}

    \bigskip

    \alert{Mistura rápida = pequeno tempo de mistura.}
\end{frame}

\begin{frame}{Lacuna Espectral}
    AJUSTAR
    Seja

    \[
        1=\mu_1>\mu_2\geq\cdots\geq\mu_n
    \]

    os autovalores da matriz de transição \(P\).

    \bigskip

    Definimos a \textbf{lacuna espectral} por

    \[
        \gamma = 1-\mu_2.
    \]

    \bigskip

    \textbf{Intuição:}

    \begin{itemize}
        \item \(\mu_2\) mede quanto da informação inicial ainda permanece.
        \item \(\mu_2\) próximo de \(1\) $\Rightarrow$ mistura lenta.
        \item \(\mu_2\) pequeno $\Rightarrow$ mistura rápida.
    \end{itemize}

    \bigskip

    \[
        \boxed{\text{Gap grande } \Rightarrow \text{ mistura rápida}}
    \]

\end{frame}


\begin{frame}{Tempo de Mistura e Lacuna Espectral}
    AJUSTAR
    \textbf{Teorema}
    \bigskip
    Para o passeio aleatório preguiçoso,

    \[
        \tau_{mix}(\varepsilon)
        =
        O\!\left(
            \frac{1}{\gamma}
            \log n
        \right).
    \]

    \bigskip

    Como

    \[
        \gamma = 1-\mu_2,
    \]

    temos:

    \bigskip

    \[
        \gamma \uparrow
        \quad \Longrightarrow \quad
        \tau_{mix} \downarrow
    \]

    \[
        \gamma \downarrow
        \quad \Longrightarrow \quad
        \tau_{mix} \uparrow
    \]

    \bigskip

    \alert{
        Quanto maior a lacuna espectral,
        mais rapidamente o passeio converge para a distribuição estacionária.
    }

\end{frame}



\begin{frame}{Autovalores do Passeio Preguiçoso}
    AJUSTAR
    Sabemos que

    \[
        P=\frac12 I+\frac1{2d}A.
    \]

    Se

    \[
        Av_i=\lambda_i v_i,
    \]

    então

    \[
        Pv_i
        =
        \frac12 v_i
        +
        \frac1{2d}Av_i
        =
        \frac12 v_i
        +
        \frac{\lambda_i}{2d}v_i.
    \]

    Logo

    \[
        Pv_i
        =
        \left(
            \frac12+\frac{\lambda_i}{2d}
        \right)v_i.
    \]

    \bigskip

    Portanto,

    \[
        \boxed{
            \mu_i
            =
            \frac12+\frac{\lambda_i}{2d}
        }
    \]

    \bigskip

    Os autovalores do passeio são determinados pelos autovalores do grafo.

\end{frame}
