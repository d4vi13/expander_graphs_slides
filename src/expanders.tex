\begin{frame}{Constante de Cheeger ou RazĂŁo de ExpansĂŁo}
  Precisamos de outra forma de medir isso:  
  \visible<2->{%
    \begin{definition}[Constante de Cheeger]
      Seja $G = (V, E)$ um grafo. A constante de Cheeger de $G$, denotada por $h(G)$, ĂŠ definida por
      \[
        h(G) = \min_{S\subsetneq V }\frac{|\partial S|}{\min(|S|, |V|\setminus|S|)}
      \]
    \end{definition}
  }
\end{frame}

\begin{frame}
  \begin{example}
    Considere o grafo completo $K_n$ e $S \subset V$ um subconjunto com m
    vertĂ­ces, temos que:

    \[
      |\partial S| = m(n-m)
    \]

    EntĂŁo:

    \[
      \frac{|\partial S|}{\min(|S|,|V|\setminus|S|)} =\frac{m(n-m)}{\min(m, n-m)} = \max(m , n - m).
    \]

    O valor que minimiza a expressĂŁo ĂŠ $m=\frac{n}2$, logo teremos que:

    \[
        h(K_n) = \frac{n}2
    \]

    \vspace{5pt}

  \end{example}
\end{frame}

\subsection{ExpansĂŁo e a lacuna espectral}

\begin{frame}{InequaĂ§ĂŁo de Cheeger}
\begin{theorem}[InequaĂ§ĂŁo de Cheeger]
\label{thm:cheeger}
Seja $G$ um grafo $d$-regular e $\lambda_1\ge \lambda_2 \ge \dots \ge \lambda_n$ seu espectro. EntĂŁo
\[
\frac{d-\lambda_2}{2}\leq h(G) \leq \sqrt{2d(d-\lambda_2)}.
\]
\footnotetext{Lembre que $d-\lambda_2 $ ĂŠ a lacuna espectral}
\end{theorem}
\vspace{10pt}
\visible<2->{
  \begin{center}
  \large A constante de cheeger estĂĄ intimamente relacionada a lacuna espectral
  \end{center}
}
\end{frame}

\subsection{ExpansĂŁo e o tempo de mistura}

\begin{frame}
  Seja $G=(V,E)$ um grafo $d$-regular conexo e $\lambda_1\ge \lambda_2 \ge \dots
  \ge \lambda_n$ seu espectro, considere um passeio aleatĂłrio
  preguiĂ§oso em G com matriz de transiĂ§ĂŁo P, lembre que lacuna espectral de P ĂŠ a seguinte:

\[
\gamma = \frac{d-\lambda_2}{2d}
\]

\visible<2->{
EntĂŁo substituindo na inequaĂ§ĂŁo de Cheeger:
\[
d\gamma  \leq h(G) \leq \sqrt{(2d)^2\gamma} \implies 
d\gamma  \leq h(G) \leq 2d\sqrt{\gamma}
\]
}

\visible<3->{
Logo a constante de Cheeger estĂĄ relacionada a lacuna espectral do passeio
}
\end{frame}

\begin{frame}{Relemebrando...}
  \begin{center}
  A lacuna espectral do passeio determina o tempo de mistura:
\[
  \mathcal{O}\!\left(\frac{1}{\gamma}\log n\right).
\]

Portanto a constante indiretamente irĂĄ defninir o tempo de mistura.

\vspace{20pt}
\visible<2->{
\large \textcolor{red}{A constante de cheeger ĂŠ inversamente
proporcional ao tempo de mistura}
}
  \end{center}
\end{frame}
